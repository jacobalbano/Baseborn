package ifrit
{
	import adobe.utils.XMLUI;
	import flash.display.DisplayObject;
	import flash.utils.Timer;
	
	public final class Rules
	{
		private static var dY:Number // Vertical displacement
		private static var dX:Number // Horizontal displacement
		
		
		private static var ivY:Number; // Initial vertical velocity
		private static var ivX:Number; // Initial horizontal velocity
		
		private static var time:Timer = new Timer(100);
		private static var t:Number; // time
		
		public static const gravity:Number = 1;
		
		public function Rules()
		{
			ivX = 7;
			ivY = 0;
			time.start();
		}
		
		public static function verticalDisplacement():Number
		{
			
			t = time.currentCount;
			ivY = 0;
			
			//trace(t);
			
			dY = (ivY * t) + (.5 * (gravity * (t * t)));
			
			return dY;
		}
		
		public static function horizontalDisplacement():Number
		{
			
			t = time.currentCount + 1;
			ivX = 7;
			
			trace(t);
			
			dX = ivX * t;
			
			return dX;
		}
		
		public static function resetTime():void
		{
			time.stop();
			time.reset();
		}
	
	}

}
